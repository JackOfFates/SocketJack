// ======== VISUAL QUERY BUILDER ========
var qbActive=false;
var qbNodes=[];
var qbConnections=[];
var qbNextId=1;
var qbDragNode=null,qbDragOfsX=0,qbDragOfsY=0;
var qbPanX=0,qbPanY=0,qbPanning=false,qbPanStartX=0,qbPanStartY=0;
var qbConnecting=false,qbConnSrc=null,qbConnSrcPort=null;
var qbTableColumns={};

function openQueryBuilder(){
 qbActive=true;
 if(dsnActive)closeDesigner();
 if(apiActive){apiActive=false;closeApiEditor();document.getElementById('apiCreatorView').classList.remove('active')}
 document.getElementById('queryView').classList.add('hidden');
 document.getElementById('qbView').classList.add('active');
 // Sync db select
 var dbSel=document.getElementById('qbDbSelect');
 dbSel.innerHTML=document.getElementById('dbSelect').innerHTML;
 dbSel.value=currentDb;
 qbLoadTables();
 qbRender();
}
function closeQueryBuilder(){
 qbActive=false;
 document.getElementById('queryView').classList.remove('hidden');
 document.getElementById('qbView').classList.remove('active');
}

function qbLoadTables(){
 var db=document.getElementById('qbDbSelect').value||currentDb;
 fetch(BASE+'/api/tables?db='+encodeURIComponent(db)).then(function(r){return r.json()}).then(function(d){
  // Add table items to palette dynamically
  var palette=document.getElementById('qbPalette');
  // Remove old dynamic table items
  var old=palette.querySelectorAll('.qb-dyn-table');
  for(var i=0;i<old.length;i++)old[i].parentNode.removeChild(old[i]);
  // Find the first palette group to insert after
  var firstGroup=palette.querySelector('.qb-palette-group');
  var tables=d.tables||[];
  tables.forEach(function(tbl){
   var item=document.createElement('div');
   item.className='qb-palette-item qb-dyn-table';
   item.draggable=true;
   item.dataset.nodeType='table';
   item.dataset.tableName=tbl.name;
   item.innerHTML='<span class="pi-icon">&#128203;</span><span class="pi-label">'+escapeHtml(tbl.name)+'</span><span class="pi-desc">'+tbl.rowCount+'r</span>';
   firstGroup.parentNode.insertBefore(item,firstGroup.nextSibling);
   qbSetupDrag(item);
   // Preload columns
   qbFetchColumns(db,tbl.name);
  });
 });
}

function qbFetchColumns(db,table){
 fetch(BASE+'/api/columns?db='+encodeURIComponent(db)+'&table='+encodeURIComponent(table)).then(function(r){return r.json()}).then(function(d){
  qbTableColumns[table]=(d.columns||[]).map(function(c){return{name:c.name,type:c.type}});
 });
}

// ---- Node model ----
function qbCreateNode(type,x,y,opts){
 var node={id:qbNextId++,type:type,x:x,y:y,w:200,h:0,data:{},inputs:[],outputs:[]};
 opts=opts||{};
 switch(type){
  case 'table':
   node.data.table=opts.tableName||'';
   node.data.alias='';
   node.outputs=[{id:'rows',label:'Rows'}];
   break;
  case 'select':
   node.data.columns='*';
   node.data.distinct=false;
   node.inputs=[{id:'source',label:'Source'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'where':
   node.data.condition='';
   node.inputs=[{id:'source',label:'Source'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'orderby':
   node.data.columns='';
   node.data.direction='ASC';
   node.inputs=[{id:'source',label:'Source'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'groupby':
   node.data.columns='';
   node.data.having='';
   node.inputs=[{id:'source',label:'Source'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'join':
   node.data.joinType='INNER JOIN';
   node.data.onCondition='';
   node.inputs=[{id:'left',label:'Left'},{id:'right',label:'Right'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'top':
   node.data.count='100';
   node.inputs=[{id:'source',label:'Source'}];
   node.outputs=[{id:'result',label:'Result'}];
   break;
  case 'insert':
   node.data.columns='';
   node.data.values='';
   node.inputs=[{id:'target',label:'Target'}];
   node.outputs=[];
   break;
  case 'update':
   node.data.set='';
   node.inputs=[{id:'target',label:'Target'},{id:'filter',label:'Where'}];
   node.outputs=[];
   break;
  case 'delete':
   node.inputs=[{id:'target',label:'Target'},{id:'filter',label:'Where'}];
   node.outputs=[];
   break;
     case 'aggregate':
      node.data.func='COUNT';
      node.data.column='*';
      node.data.alias='';
      node.inputs=[{id:'source',label:'Source'}];
      node.outputs=[{id:'result',label:'Result'}];
      break;
     case 'output':
      node.data.label='Output';
      node.inputs=[{id:'result',label:'Result'}];
      node.outputs=[];
      break;
    }
    qbNodes.push(node);
    return node;
  }

// ---- Palette drag ----
function qbSetupDrag(el){
 el.addEventListener('dragstart',function(e){
  e.dataTransfer.setData('text/plain',JSON.stringify({type:el.dataset.nodeType,tableName:el.dataset.tableName||''}));
  e.dataTransfer.effectAllowed='copy';
 });
}
// Setup built-in palette items
document.querySelectorAll('.qb-palette-item[draggable]').forEach(function(el){qbSetupDrag(el)});

var qbCanvasWrap=document.getElementById('qbCanvasWrap');
qbCanvasWrap.addEventListener('dragover',function(e){e.preventDefault();e.dataTransfer.dropEffect='copy'});
qbCanvasWrap.addEventListener('drop',function(e){
 e.preventDefault();
 var data;try{data=JSON.parse(e.dataTransfer.getData('text/plain'))}catch(x){return}
 var rect=qbCanvasWrap.getBoundingClientRect();
 var x=e.clientX-rect.left-qbPanX;
 var y=e.clientY-rect.top-qbPanY;
 qbCreateNode(data.type,x,y,{tableName:data.tableName});
 qbRender();
 qbUpdateSql();
});

// ---- Panning ----
qbCanvasWrap.addEventListener('mousedown',function(e){
 if(e.target===qbCanvasWrap||e.target.id==='qbSvg'){
  qbPanning=true;qbPanStartX=e.clientX-qbPanX;qbPanStartY=e.clientY-qbPanY;
  qbCanvasWrap.style.cursor='grabbing';e.preventDefault();
 }
});
document.addEventListener('mousemove',function(e){
 if(qbPanning){
  qbPanX=e.clientX-qbPanStartX;qbPanY=e.clientY-qbPanStartY;
  qbApplyPan();
 }
 if(qbDragNode){
  var rect=qbCanvasWrap.getBoundingClientRect();
  qbDragNode.x=e.clientX-rect.left-qbPanX-qbDragOfsX;
  qbDragNode.y=e.clientY-rect.top-qbPanY-qbDragOfsY;
  qbRender();
 }
 if(qbConnecting){
  qbRenderTempLine(e);
 }
});
document.addEventListener('mouseup',function(e){
 if(qbPanning){qbPanning=false;qbCanvasWrap.style.cursor=''}
 if(qbDragNode){qbDragNode=null}
 if(qbConnecting){qbConnecting=false;qbConnSrc=null;qbRenderConnections()}
});
function qbApplyPan(){
 document.getElementById('qbNodes').style.transform='translate('+qbPanX+'px,'+qbPanY+'px)';
 document.getElementById('qbSvg').style.transform='translate('+qbPanX+'px,'+qbPanY+'px)';
}

// ---- Rendering ----
function qbRender(){
 var container=document.getElementById('qbNodes');
 container.innerHTML='';
 qbNodes.forEach(function(node){
  var div=document.createElement('div');
  div.style.cssText='position:absolute;left:'+node.x+'px;top:'+node.y+'px;width:'+node.w+'px;background:#2d2d2d;border:1px solid #3c3c3c;border-radius:6px;pointer-events:auto;user-select:none;box-shadow:0 2px 8px rgba(0,0,0,.3)';
  div.dataset.nodeId=node.id;
  // Header
  var hdr=document.createElement('div');
  hdr.style.cssText='padding:6px 10px;background:#333;border-radius:5px 5px 0 0;display:flex;align-items:center;gap:6px;cursor:move;border-bottom:1px solid #3c3c3c';
  hdr.innerHTML=qbNodeIcon(node.type)+'<span style="flex:1;font-size:12px;font-weight:600;color:'+qbNodeColor(node.type)+'">'+(node.type==='table'?(node.data.table||'Table'):node.type.toUpperCase())+'</span>';
  var closeBtn=document.createElement('span');
  closeBtn.style.cssText='color:#808080;cursor:pointer;font-size:14px';closeBtn.textContent='\u00d7';
  closeBtn.addEventListener('click',function(e){e.stopPropagation();qbRemoveNode(node.id)});
  hdr.appendChild(closeBtn);
  // Drag header
  hdr.addEventListener('mousedown',function(e){
   if(e.target===closeBtn)return;
   e.stopPropagation();
   qbDragNode=node;
   var rect=qbCanvasWrap.getBoundingClientRect();
   qbDragOfsX=e.clientX-rect.left-qbPanX-node.x;
   qbDragOfsY=e.clientY-rect.top-qbPanY-node.y;
  });
  div.appendChild(hdr);
  // Body
  var body=document.createElement('div');
  body.style.cssText='padding:8px 10px;font-size:11px';
  body.innerHTML=qbNodeBody(node);
  div.appendChild(body);
  // Ports
  var ports=document.createElement('div');
  ports.style.cssText='display:flex;justify-content:space-between;padding:4px 6px 6px';
  // Input ports (left)
  var leftPorts=document.createElement('div');
  node.inputs.forEach(function(inp){
   var p=document.createElement('div');
   p.style.cssText='display:flex;align-items:center;gap:4px;margin-bottom:2px;cursor:pointer';
   p.innerHTML='<span style="width:10px;height:10px;border-radius:50%;background:#569cd6;display:inline-block;border:2px solid #1e1e1e"></span><span style="color:#808080;font-size:10px">'+inp.label+'</span>';
   p.dataset.nodeId=node.id;p.dataset.portId=inp.id;p.dataset.portDir='in';
   p.addEventListener('mouseup',function(e){
    e.stopPropagation();
    if(qbConnecting&&qbConnSrc){
     qbAddConnection(qbConnSrc.nodeId,qbConnSrc.portId,node.id,inp.id);
     qbConnecting=false;qbConnSrc=null;
     qbRenderConnections();qbUpdateSql();
    }
   });
   leftPorts.appendChild(p);
  });
  ports.appendChild(leftPorts);
  // Output ports (right)
  var rightPorts=document.createElement('div');
  rightPorts.style.cssText='text-align:right';
  node.outputs.forEach(function(out){
   var p=document.createElement('div');
   p.style.cssText='display:flex;align-items:center;gap:4px;margin-bottom:2px;cursor:pointer;justify-content:flex-end';
   p.innerHTML='<span style="color:#808080;font-size:10px">'+out.label+'</span><span style="width:10px;height:10px;border-radius:50%;background:#4ec9b0;display:inline-block;border:2px solid #1e1e1e"></span>';
   p.dataset.nodeId=node.id;p.dataset.portId=out.id;p.dataset.portDir='out';
   p.addEventListener('mousedown',function(e){
    e.stopPropagation();
    qbConnecting=true;
    qbConnSrc={nodeId:node.id,portId:out.id};
   });
   rightPorts.appendChild(p);
  });
  ports.appendChild(rightPorts);
  div.appendChild(ports);
  container.appendChild(div);
  node.h=div.offsetHeight;
 });
 qbRenderConnections();
}

function qbNodeIcon(type){
 var icons={table:'&#128203;',select:'&#128269;',where:'&#128270;',orderby:'&#8645;',groupby:'&#128202;',join:'&#128279;',top:'#',insert:'&#10010;',update:'&#9998;','delete':'&#128465;',aggregate:'&#931;',output:'&#9654;'};
 return '<span style="font-size:14px">'+(icons[type]||'&#9679;')+'</span>';
}
function qbNodeColor(type){
 var colors={table:'#4ec9b0',select:'#569cd6',where:'#dcdcaa',orderby:'#c586c0',groupby:'#c586c0',join:'#ce9178',top:'#b5cea8',insert:'#6a9955',update:'#d7ba7d','delete':'#f44747',aggregate:'#c586c0',output:'#569cd6'};
 return colors[type]||'#d4d4d4';
}

function qbSetNodeData(nodeId,key,val){
 for(var i=0;i<qbNodes.length;i++){
  if(qbNodes[i].id===nodeId){qbNodes[i].data[key]=val;return}
 }
}

function qbRemoveNode(id){
 qbNodes=qbNodes.filter(function(n){return n.id!==id});
 qbConnections=qbConnections.filter(function(c){return c.fromNode!==id&&c.toNode!==id});
 qbRender();qbUpdateSql();
}

function qbClear(){
 if(qbNodes.length>0&&!confirm('Clear all nodes?'))return;
 qbNodes=[];qbConnections=[];qbNextId=1;
 qbRender();qbUpdateSql();
}

// ---- Connections ----
function qbAddConnection(fromNode,fromPort,toNode,toPort){
 // Remove existing connection to same input
 qbConnections=qbConnections.filter(function(c){return !(c.toNode===toNode&&c.toPort===toPort)});
 qbConnections.push({fromNode:fromNode,fromPort:fromPort,toNode:toNode,toPort:toPort});
}

function qbGetPortPos(nodeId,portId,dir){
 var container=document.getElementById('qbNodes');
 var nodeEl=container.querySelector('[data-node-id="'+nodeId+'"]');
 if(!nodeEl)return{x:0,y:0};
 var portEl=nodeEl.querySelector('[data-port-id="'+portId+'"][data-port-dir="'+dir+'"]');
 if(!portEl){
  var nx=parseFloat(nodeEl.style.left)||0;
  var ny=parseFloat(nodeEl.style.top)||0;
  return{x:dir==='out'?nx+200:nx,y:ny+20};
 }
 var dot=portEl.querySelector('span');
 if(!dot)dot=portEl;
 var nr=nodeEl.getBoundingClientRect();
 var dr=dot.getBoundingClientRect();
 var cx=dr.left-nr.left+dr.width/2+parseFloat(nodeEl.style.left);
 var cy=dr.top-nr.top+dr.height/2+parseFloat(nodeEl.style.top);
 return{x:cx,y:cy};
}

function qbRenderConnections(){
 var svg=document.getElementById('qbSvg');
 svg.innerHTML='';
 var wrap=document.getElementById('qbCanvasWrap');
 svg.setAttribute('width',wrap.scrollWidth||4000);
 svg.setAttribute('height',wrap.scrollHeight||4000);
 svg.style.overflow='visible';
 qbConnections.forEach(function(conn){
  var from=qbGetPortPos(conn.fromNode,conn.fromPort,'out');
  var to=qbGetPortPos(conn.toNode,conn.toPort,'in');
  var dx=Math.abs(to.x-from.x)*0.5;
  var path=document.createElementNS('http://www.w3.org/2000/svg','path');
  path.setAttribute('d','M'+from.x+','+from.y+' C'+(from.x+dx)+','+from.y+' '+(to.x-dx)+','+to.y+' '+to.x+','+to.y);
  path.setAttribute('stroke','#569cd6');path.setAttribute('stroke-width','2');path.setAttribute('fill','none');path.setAttribute('opacity','0.6');
  svg.appendChild(path);
 });
}
function qbRenderTempLine(e){
 var svg=document.getElementById('qbSvg');
 // Remove old temp line
 var old=svg.querySelector('.qb-temp-line');if(old)old.remove();
 if(!qbConnSrc)return;
 var from=qbGetPortPos(qbConnSrc.nodeId,qbConnSrc.portId,'out');
 var rect=qbCanvasWrap.getBoundingClientRect();
 var tx=e.clientX-rect.left-qbPanX;
 var ty=e.clientY-rect.top-qbPanY;
 var dx=Math.abs(tx-from.x)*0.5;

function qbFindNode(id){
 for(var i=0;i<qbNodes.length;i++){if(qbNodes[i].id===id)return qbNodes[i]}
 return null;
}
function qbGetConnectedSource(nodeId,portId){
 for(var i=0;i<qbConnections.length;i++){
  var c=qbConnections[i];
  if(c.toNode===nodeId&&c.toPort===portId)return qbFindNode(c.fromNode);
 }
 return null;
}
function qbGetChain(nodeId,portId){
 // Walk backwards through connections to find the chain
 var chain=[];
 var current=qbFindNode(nodeId);
 if(!current)return chain;
 chain.unshift(current);
 var src=qbGetConnectedSource(nodeId,portId||current.inputs[0]?.id);
 var visited={};visited[nodeId]=true;
 while(src&&!visited[src.id]){
  visited[src.id]=true;
  chain.unshift(src);
  if(src.inputs.length>0){
   src=qbGetConnectedSource(src.id,src.inputs[0].id);
  } else break;
 }
 return chain;
}

function qbGenerateSql(){
 if(qbNodes.length===0)return '';
 // Find terminal nodes (nodes with no outgoing connections used)
 var hasOutgoing={};
 qbConnections.forEach(function(c){hasOutgoing[c.fromNode]=true});
 // Prefer explicit Output nodes as the terminal
 var outputNodes=qbNodes.filter(function(n){return n.type==='output'});
 var terminals;
 if(outputNodes.length>0){
  terminals=outputNodes;
 } else {
  terminals=qbNodes.filter(function(n){return !hasOutgoing[n.id]||n.outputs.length===0});
 }
 if(terminals.length===0)terminals=[qbNodes[qbNodes.length-1]];
 // Use the first terminal
 var terminal=terminals[0];
 return qbBuildFromTerminal(terminal);
}

function qbBuildFromTerminal(node){
 // Get the full chain leading to this terminal
 var mainPort=node.inputs.length>0?node.inputs[0].id:null;
 var chain=mainPort?qbGetChain(node.id,mainPort):[node];
 // Find table nodes, select nodes, etc.
 var tables=[],selectNode=null,whereNode=null,orderNode=null,groupNode=null,topNode=null,joinNode=null,insertNode=null,updateNode=null,deleteNode=null,aggregateNode=null;
 chain.forEach(function(n){
  switch(n.type){
   case 'table':tables.push(n);break;
   case 'select':selectNode=n;break;
   case 'where':whereNode=n;break;
   case 'orderby':orderNode=n;break;
   case 'groupby':groupNode=n;break;
   case 'top':topNode=n;break;
   case 'join':joinNode=n;break;
   case 'insert':insertNode=n;break;
   case 'update':updateNode=n;break;
   case 'delete':deleteNode=n;break;
    case 'aggregate':aggregateNode=n;break;
    case 'output':break;
   }
 });
 // For join, get right table
 if(joinNode){
  var rightSrc=qbGetConnectedSource(joinNode.id,'right');
  if(rightSrc&&rightSrc.type==='table'&&tables.indexOf(rightSrc)<0)tables.push(rightSrc);
 }
 // For update/delete, check for where connection
 if(updateNode){
  var wSrc=qbGetConnectedSource(updateNode.id,'filter');
  if(wSrc&&wSrc.type==='where')whereNode=wSrc;
 }
 if(deleteNode){
  var wSrc2=qbGetConnectedSource(deleteNode.id,'filter');
  if(wSrc2&&wSrc2.type==='where')whereNode=wSrc2;
 }

 var table0=tables.length>0?tables[0]:null;
 var tblName=table0?('['+table0.data.table+']'+(table0.data.alias?' '+table0.data.alias:'')):'[?]';

 // INSERT
 if(insertNode&&table0){
  var iCols=insertNode.data.columns?insertNode.data.columns.trim():'';
  var iVals=insertNode.data.values?insertNode.data.values.trim():'';
  var sql='INSERT INTO ['+table0.data.table+']';
  if(iCols)sql+=' ('+iCols+')';
  sql+='\nVALUES ('+iVals+')';
  return sql;
 }
 // UPDATE
 if(updateNode&&table0){
  var sql='UPDATE ['+table0.data.table+']\nSET '+(updateNode.data.set||'?');
  if(whereNode&&whereNode.data.condition)sql+='\nWHERE '+whereNode.data.condition;
  return sql;
 }
 // DELETE
 if(deleteNode&&table0){
  var sql='DELETE FROM ['+table0.data.table+']';
  if(whereNode&&whereNode.data.condition)sql+='\nWHERE '+whereNode.data.condition;
  return sql;
 }

 // SELECT query
 var cols='*';
 if(selectNode&&selectNode.data.columns&&selectNode.data.columns.trim())cols=selectNode.data.columns.trim();
 if(aggregateNode){
  var aggExpr=aggregateNode.data.func+'('+(aggregateNode.data.column||'*')+')';
  if(aggregateNode.data.alias)aggExpr+=' AS '+aggregateNode.data.alias;
  if(cols==='*')cols=aggExpr;
  else cols+=', '+aggExpr;
 }
 var distinct=(selectNode&&selectNode.data.distinct)?'DISTINCT ':'';
 var topStr='';
 if(topNode&&topNode.data.count)topStr='TOP '+topNode.data.count+' ';

 var sql='SELECT '+topStr+distinct+cols;
 if(table0){
  sql+='\nFROM '+tblName;
 }
 if(joinNode&&tables.length>1){
  var rightTbl=tables[1];
  sql+='\n'+joinNode.data.joinType+' ['+rightTbl.data.table+']'+(rightTbl.data.alias?' '+rightTbl.data.alias:'');
  if(joinNode.data.onCondition)sql+=' ON '+joinNode.data.onCondition;
 }
 if(whereNode&&whereNode.data.condition)sql+='\nWHERE '+whereNode.data.condition;
 if(groupNode&&groupNode.data.columns){
  sql+='\nGROUP BY '+groupNode.data.columns;
  if(groupNode.data.having)sql+='\nHAVING '+groupNode.data.having;
 }
 if(orderNode&&orderNode.data.columns)sql+='\nORDER BY '+orderNode.data.columns+' '+orderNode.data.direction;
 return sql;
}

function qbHighlight(sql){
 var keywords=['SELECT','FROM','WHERE','ORDER BY','GROUP BY','HAVING','INSERT','INTO','VALUES','UPDATE','SET','DELETE','JOIN','INNER JOIN','LEFT JOIN','RIGHT JOIN','FULL OUTER JOIN','CROSS JOIN','ON','AND','OR','NOT','IN','LIKE','BETWEEN','IS','NULL','AS','DISTINCT','TOP','ASC','DESC','COUNT','SUM','AVG','MIN','MAX'];
 var escaped=escapeHtml(sql);
 keywords.forEach(function(kw){
  var re=new RegExp('\\b('+kw.replace(/ /g,'\\s+')+ ')\\b','gi');
  escaped=escaped.replace(re,'<span class="kw">$1</span>');
 });
 // Highlight strings
 escaped=escaped.replace(/'([^']*)'/g,'<span class="str">\'$1\'</span>');
 // Highlight numbers
 escaped=escaped.replace(/\b(\d+)\b/g,'<span class="num">$1</span>');
 return escaped;
}

function qbRunQuery(){
 var sql=qbGenerateSql();
 if(!sql){alert('No query to run. Add nodes first.');return}
 var db=document.getElementById('qbDbSelect').value||currentDb;
 var bar=document.getElementById('qbResultBar');
 bar.classList.add('show');bar.textContent='Executing...';
 fetch(BASE+'/api/query',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({query:sql,database:db})})
 .then(function(r){return r.json()}).then(function(d){
  if(d.error){bar.textContent='Error: '+d.error;bar.style.color='#f44747';return}
  if(d.columns){
   var rc=(d.rows||[]).length;
   bar.textContent='Query returned '+rc+' row'+(rc!==1?'s':'')+'.'+(d.rowsAffected?' Rows affected: '+d.rowsAffected:'');
   bar.style.color='#b5cea8';
   // Also put result in main editor for viewing
   lastGridHtml=buildResultGrid(d);
   messages.push({time:new Date().toLocaleTimeString(),text:'(Query Builder) '+rc+' rows returned'});
  } else {
   bar.textContent=(d.message||'Query executed.')+(d.rowsAffected?' Rows affected: '+d.rowsAffected:'');
   bar.style.color='#b5cea8';
   messages.push({time:new Date().toLocaleTimeString(),text:'(Query Builder) '+d.message});
  }
 }).catch(function(e){bar.textContent='Error: '+e.message;bar.style.color='#f44747'});
}

function buildResultGrid(d){
 var html='<table class="results-grid"><thead><tr>';
 d.columns.forEach(function(c){html+='<th>'+escapeHtml(c)+'</th>'});
 html+='</tr></thead><tbody>';
 (d.rows||[]).forEach(function(row){
  html+='<tr>';
  row.forEach(function(v){
   if(v===null)html+='<td class="null-val">NULL</td>';
   else html+='<td>'+escapeHtml(String(v))+'</td>';
  });
  html+='</tr>';
 });
 html+='</tbody></table>';
 return html;
}
