// ======== API CREATOR ========
var apiActive=false;
var apiEndpoints=[];

function openApiCreator(){
 apiActive=true;
 if(dsnActive)closeDesigner();
 if(qbActive){qbActive=false;document.getElementById('qbView').classList.remove('active')}
 document.getElementById('queryView').classList.add('hidden');
 document.getElementById('apiCreatorView').classList.add('active');
 loadApiEndpoints();
}
function closeApiCreator(){
 apiActive=false;
 closeApiEditor();
 document.getElementById('queryView').classList.remove('hidden');
 document.getElementById('apiCreatorView').classList.remove('active');
}
function loadApiEndpoints(){
 fetch(BASE+'/api/endpoints/list').then(function(r){return r.json()}).then(function(d){
  apiEndpoints=d.endpoints||[];
  renderApiList();
 });
}
function renderApiList(){
 var list=document.getElementById('apiList');
 if(apiEndpoints.length===0){
  list.innerHTML='<div class="api-empty"><div class="icon">&#128268;</div>No API endpoints yet.<br>Click "+ New Endpoint" to create one.</div>';
  return;
 }
 var html='';
 apiEndpoints.forEach(function(ep){
  html+='<div class="api-card" onclick="openApiEditor(\''+ep.id+'\')"><div class="api-name"><span class="method-badge '+ep.httpMethod.toLowerCase()+'">'
   +escapeHtml(ep.httpMethod)+'</span> '+escapeHtml(ep.name)+'</div>'
   +'<div class="api-route">'+escapeHtml(ep.route)+'</div>'
   +'<div class="api-meta"><span>DB: '+escapeHtml(ep.database)+'</span><span>Format: '+escapeHtml(ep.responseFormat)+'</span>'
   +(ep.enabled?'<span style="color:#16c784">\u25CF Active</span>':'<span style="color:#808080">\u25CB Disabled</span>')+'</div></div>';
 });
 list.innerHTML=html;
}
function openApiEditor(id){
 var isEdit=!!id;
 document.getElementById('apiList').style.display='none';
 document.getElementById('apiEditor').classList.add('active');
 document.getElementById('apiDeleteBtn').style.display=isEdit?'':'none';
 document.getElementById('apiTestSection').classList.remove('show');
 // Populate database select from main db selector
 var dbSel=document.getElementById('apiDatabase');
 dbSel.innerHTML=document.getElementById('dbSelect').innerHTML;
 if(id){
  var ep=null;
  for(var i=0;i<apiEndpoints.length;i++){if(apiEndpoints[i].id===id){ep=apiEndpoints[i];break}}
  if(ep){
   document.getElementById('apiEditId').value=ep.id;
   document.getElementById('apiName').value=ep.name;
   document.getElementById('apiRoute').value=ep.route;
   document.getElementById('apiMethod').value=ep.httpMethod;
   document.getElementById('apiDatabase').value=ep.database;
   document.getElementById('apiSqlQuery').value=ep.sqlQuery;
   document.getElementById('apiFormat').value=ep.responseFormat;
   document.getElementById('apiContentType').value=ep.contentType||'';
  }
 } else {
  document.getElementById('apiEditId').value='';
  document.getElementById('apiName').value='';
  document.getElementById('apiRoute').value='/api/';
  document.getElementById('apiMethod').value='GET';
  document.getElementById('apiDatabase').value=currentDb;
  document.getElementById('apiSqlQuery').value='';
  document.getElementById('apiFormat').value='json';
  document.getElementById('apiContentType').value='';
 }
 onApiFormatChange();
}
function closeApiEditor(){
 document.getElementById('apiList').style.display='';
 document.getElementById('apiEditor').classList.remove('active');
}
function onApiFormatChange(){
 document.getElementById('apiContentTypeField').style.display=document.getElementById('apiFormat').value==='binary'?'':'none';
}
function saveApiEndpoint(){
 var payload={
  id:document.getElementById('apiEditId').value||'',
  name:document.getElementById('apiName').value,
  route:document.getElementById('apiRoute').value,
  httpMethod:document.getElementById('apiMethod').value,
  database:document.getElementById('apiDatabase').value,
  sqlQuery:document.getElementById('apiSqlQuery').value,
  responseFormat:document.getElementById('apiFormat').value,
  contentType:document.getElementById('apiContentType').value,
  enabled:'true'
 };
 if(!payload.name||!payload.route||!payload.sqlQuery){alert('Name, route, and SQL query are required.');return}
 fetch(BASE+'/api/endpoints/save',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(payload)})
 .then(function(r){return r.json()}).then(function(r){
  if(r.error){alert(r.error);return}
  closeApiEditor();
  loadApiEndpoints();
 });
}
function deleteApiEndpoint(){
 var id=document.getElementById('apiEditId').value;
 if(!id||!confirm('Delete this API endpoint?'))return;
 fetch(BASE+'/api/endpoints/delete',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({id:id})})
 .then(function(r){return r.json()}).then(function(r){
  if(r.error){alert(r.error);return}
  closeApiEditor();
  loadApiEndpoints();
 });
}
function testApiEndpoint(){
 var section=document.getElementById('apiTestSection');
 section.classList.toggle('show');
 document.getElementById('apiTestUrl').textContent=location.origin+document.getElementById('apiRoute').value;
 document.getElementById('apiTestResult').style.display='none';
}
function runApiTest(){
 var paramsStr=document.getElementById('apiTestParams').value.trim();
 var payload={
  route:document.getElementById('apiRoute').value,
  httpMethod:document.getElementById('apiMethod').value,
  database:document.getElementById('apiDatabase').value,
  sqlQuery:document.getElementById('apiSqlQuery').value,
  responseFormat:document.getElementById('apiFormat').value,
  contentType:document.getElementById('apiContentType').value,
  parameters:paramsStr||'{}'
 };
 document.getElementById('apiTestResult').style.display='block';
 document.getElementById('apiTestResult').textContent='Executing...';
 fetch(BASE+'/api/endpoints/test',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(payload)})
 .then(function(r){return r.text()}).then(function(text){
  var el=document.getElementById('apiTestResult');
  try{el.textContent=JSON.stringify(JSON.parse(text),null,2)}catch(e){el.textContent=text}
 }).catch(function(e){
  document.getElementById('apiTestResult').textContent='Error: '+e.message;
 });
}
